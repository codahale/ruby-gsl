#!/usr/local/bin/ruby


require "GSL"
include GSL

STDERR.puts "Running tests for Linalg..."

a = [ 0.18, 0.60, 0.57, 0.96,
      0.41, 0.24, 0.99, 0.58,
      0.14, 0.30, 0.97, 0.66,
      0.51, 0.13, 0.19, 0.85 ]
b = [ 1.0, 2.0, 3.0, 4.0 ]

m = Matrix.new a, 4, 4
v = Vector.new b

m, p, s = m.LU_decomp!
x = m.LU_solve p, v

puts "x = "
x.fprintf STDOUT, "%g"

STDERR.puts "\ndone."
